   LoadingTitle = "Lux Hub Loading...",
   LoadingSubtitle = "By Lux HUB üëë",
   Theme = "Ocean",
DisableRayfieldPrompts = true,
   DisableBuildWarnings = true, -- Prevents Rayfield from warning when the script has a version mismatch with the interface
   ConfigurationSaving = {
      Enabled = true,
      FileName = "LuxHubSettings"
   },
Discord = {
      Enabled = true, -- Prompt the user to join your Discord server if their executor supports it
      Invite = "hk7NZghv", -- The Discord invite code, do not include discord.gg/. E.g. discord.gg/ ABCD would be ABCD
      RememberJoins = true -- Set this to false to make them join the discord every time they load it up

-- Create Tabs
local CombatTab = Window:CreateTab("Combat", "crosshair")
local MovementTab = Window:CreateTab("Movement", "move")
local TeleportTab = Window:CreateTab("Teleport", "map")

-- Auto Aim (OP!)
local autoAimThread
CombatTab:CreateToggle({
   Name = "üí• OP Auto Aim",
   CurrentValue = false,
   Flag = "AutoAimToggle",
   Callback = function(Value)
      if Value then
         autoAimThread = task.spawn(function()
            while task.wait() do
               pcall(function()
                  local args = {
                     [1] = workspace:WaitForChild("Levels"):WaitForChild("Level14"):WaitForChild("Train1"):WaitForChild("Union")
                  }
                  game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("Fire"):FireServer(unpack(args))
                  task.wait(0.000001)
               end)
            end
         end)
      else
         if autoAimThread then
            task.cancel(autoAimThread)
         end
      end
   end,
})

-- Teleport to End
TeleportTab:CreateButton({
   Name = "üöÄ Teleport to End",
   Callback = function()
      local character = game.Players.LocalPlayer.Character
      if character then
         character:PivotTo(CFrame.new(Vector3.new(-1979, 0, -20)))
      end
   end,
})

-- WalkSpeed Modifier
MovementTab:CreateSlider({
   Name = "üèÉ WalkSpeed",
   Range = {16, 100},
   Increment = 1,
   Suffix = "Speed",
   CurrentValue = 16,
   Flag = "WalkspeedSlider",
   Callback = function(Value)
      game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = Value
   end,
})

-- Noclip Toggle
local noclip = false
MovementTab:CreateToggle({
   Name = "üöß Noclip",
   CurrentValue = false,
   Flag = "NoclipToggle",
   Callback = function(Value)
      noclip = Value
      local RunService = game:GetService("RunService")
      RunService.Stepped:Connect(function()
         if noclip then
            for _, v in pairs(game.Players.LocalPlayer.Character:GetDescendants()) do
               if v:IsA("BasePart") and v.CanCollide then
                  v.CanCollide = false
               end
            end
         end
      end)
   end,
})

-- Fly using Infinite Yield
MovementTab:CreateButton({
   Name = "üïäÔ∏è Enable Fly (Infinite Yield)",
   Callback = function()
      loadstring(game:HttpGet("https://raw.githubusercontent.com/EdgeIY/infiniteyield/master/source"))()
   end,
})

-- Load saved config
Rayfield:LoadConfiguration()
